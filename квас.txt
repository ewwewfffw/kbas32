    <script>
        // Кнопка "Нет" убегает при наведении
        function moveButton() {
            const noBtn = document.getElementById('noBtn');
            const x = Math.random() * (window.innerWidth - noBtn.offsetWidth - 40);
            const y = Math.random() * (window.innerHeight - noBtn.offsetHeight - 40);
            
            noBtn.style.position = 'fixed';
            noBtn.style.left = x + 'px';
            noBtn.style.top = y + 'px';
        }

        // Действие при прощении
        function forgive() {
            document.getElementById('title').innerText = 'Ура! Спасибо, Квас! ❤️';
            document.getElementById('message').innerText = 'Ты лучший! Я очень ценю это и исправлюсь.';
            
            // Скрываем кнопки и показываем ползунок доверия
            document.getElementById('btnBlock').style.display = 'none';
            document.getElementById('trustSection').style.display = 'block';

            // Запуск дождя из сердечек
            setInterval(createHeart, 300);
        }

        // Обновление процентов и текста ползунка
        function updateTrust(val) {
            document.getElementById('trustValue').innerText = val + '%';
            const comment = document.getElementById('trustComment');
            
            // Меняем текст комментария в зависимости от выбранного процента
            if (val <= 20) {
                comment.innerText = "Оу... Мне нужно сильно постараться, чтобы это исправить 😢";
            } else if (val <= 50) {
                comment.innerText = "Понимаю, доверие нужно заслужить заново. Я справлюсь! 🫡";
            } else if (val <= 80) {
                comment.innerText = "Спасибо за шанс! Я не подведу тебя! 🙌";
            } else {
                comment.innerText = "Вау! Квас, ты самый великодушный человек на свете! ⭐";
            }
        }

        // Генератор сердечек
        function createHeart() {
            const heart = document.createElement('div');
            heart.classList.add('heart');
            heart.innerText = '❤️';
            heart.style.left = Math.random() * 100 + "vw";
            heart.style.animationDuration = Math.random() * 2 + 2 + "s";
            document.body.appendChild(heart);

            setTimeout(() => { heart.remove(); }, 4000);
        }
    </script>
